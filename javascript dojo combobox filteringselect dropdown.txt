define([
  "dojo/store/Memory",
  "dijit/form/FilteringSelect",
], function(Memory, FileteringSelect){
	
	var combobox = new Memory({
	    data: values
	});

	combobox = new FilteringSelect({
	    id: "comboBoxUbs",
	    store: this.storeAbrangencia,
	    searchAttr: "nome",
	    style: "width:25%;position:relative; margin: 0 auto 10px auto;",
	    queryExpr: "*${0}*", //m√°gica para procurar substring
	    autoComplete: false,
	    resizable: true,
	    fetchProperties : {
	        sort : [ { attribute : "nome" } ] 
	    },
	    onChange: lang.hitch(this, this.atualizarMapa)
	}, Config.ID_LISTA_UBS);
	combobox.startup();
	combobox.focus();
	}
});
